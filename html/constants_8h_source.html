28  *
29  * This defines the maximum number of vCPUs that are supported by the VMM.

38 /*
39  * Cache Line Shift
40  *
41  * The memory manager at the moment keeps track of blocks using a cache line
42  * for performance reasons. If the cache line size is different, this value

65 /*
66  * Max Page Shift
67  *
68  * Defines the maximum page size that is supported by the VMM (not the max
69  * size supported by hardware, which is likely different). For now, this is
70  * set to a value that is likely supported by most hardware. All pages must
71  * be translated to this value, as the VMM only supports one page size.
72  *
73  * Note: defined in bits
74  */

76 #define MAX_PAGE_SHIFT (12ULL)

85  * set to a value that is likely supported by most hardware. All pages must
86  * be translated to this value, as the VMM only supports one page size.

94 /*
95  * Max Blocks
96  *
97  * This defines the maximum number of blocks that the memory manager will store
98  * for new / delete. This value will also be used to define the alloc/free
99  * bookkeeping. Note that each block is a cache line. Also note that we define
100  * the total number of blocks using the max page size, which will ensure that
101  * our mem pool is both a multiple of the cache line size, and of the page
102  * size.
103  */
104 #ifndef MAX_BLOCKS
105 #define MAX_BLOCKS (16ULL * MAX_PAGE_SIZE)
106 #endif

108 /*
109  * Convience Macros
110  *
111  * These macros provide some useful information about how the memory manager
112  * is defined. These should not be set by the compiler, but instead, are here
113  * to calculate stats about how the compiler setup the other macros.
114  */
115 #define BLOCKS_PER_PAGE (MAX_PAGE_SIZE / MAX_CACHE_LINE_SIZE)
116 #define TOTAL_NUM_PAGES (MAX_BLOCKS / BLOCKS_PER_PAGE)
