48 /*
49  * Driver Entry State Machine
50  *
51  * The driver entry has three major states that it could end up in. When the
52  * driver entry is unloaded, it means that the VMM has not been placed in
53  * memory. The loaded state means that the VMM is in memory, and relocated.
54  * In this state, symbol lookups are possible, and thus things like the VMM
55  * dump comand work. The running state means that the VMM is actually running.
56  * The goal of the state machine is to ensure that the driver keeps track of
57  * the state of the VMM, and handles its transition properly.
58  */
59 #define VMM_UNLOADED 10
60 #define VMM_LOADED 11
61 #define VMM_RUNNING 12
62 #define VMM_CORRUPT 100
