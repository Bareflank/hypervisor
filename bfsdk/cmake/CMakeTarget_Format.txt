if(ENABLE_ASTYLE AND NOT WIN32)

    message(STATUS "Code Formatting: Astyle")

    add_custom_target(
        format
        COMMAND bareflank_astyle_format.sh diff ${CMAKE_SOURCE_DIR}
        COMMAND bareflank_blank_lines.sh diff ${CMAKE_SOURCE_DIR}
        COMMENT "running code style checks: ${PROJECT_NAME} in ${CMAKE_SOURCE_DIR}"
    )

    add_custom_target(
        format-all
        COMMAND bareflank_astyle_format.sh all ${CMAKE_SOURCE_DIR}
        COMMAND bareflank_blank_lines.sh all ${CMAKE_SOURCE_DIR}
        COMMENT "running code style checks: ${PROJECT_NAME} in ${CMAKE_SOURCE_DIR}"
    )

endif()
